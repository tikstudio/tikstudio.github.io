Action 

export const TEST_DATA = 'TEST_DATA';

export function setTestData(data1, data2) {
  return {
    type: TEST_DATA, payload: {data1, data2}
  }
}
Reducer 

import {TEST_DATA} from "../actions";

const initialState = {
  data1: 0,
  data2: 0
}

function reducer(state = initialState, action) {
  switch (action.type) {
    case TEST_DATA: {
      const {data1, data2} = action.payload
      return {
        ...state,
        data1,
        data2,
      }
    }
    default: {
      return state
    }
  }
}

export default reducer

import React, {Component} from 'react';
import {setTestData} from "./store/actions";
import {connect} from "react-redux";

class App extends Component {
  handleClick = () => {
    this.props.setTestData(1, 2)
  }

  render() {
    console.log(this.props.data1);
    console.log(this.props.data2);
    return (
      <div className="App">
        <button onClick={this.handleClick}>
          test
        </button>
      </div>
    );
  }
}

const mapStateToProps = (state) => ({
  data1: state.data1,
  data2: state.data2,
});

const mapDispatchToProps = {
  setTestData
};

const AppContainer = connect(
  mapStateToProps,
  mapDispatchToProps,
)(App);

export default AppContainer;
